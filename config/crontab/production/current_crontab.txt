# Exchange Analytics Production Crontab - 日本時間（JST）版
# 24時間基本データ取得・日次・週次レポート対応
# 継続処理システム統合版
SHELL=/bin/bash
PATH=/usr/local/bin:/usr/bin:/bin
HOME=/app

# 🔄 継続処理システム（5分間隔、平日24時間稼働）
# システム初期化状態をチェックし、適切な処理を実行
*/5 * * * 1-5 cd /app && export $(cat .env | grep -v '^#' | xargs) && \
    timeout 300 python scripts/cron/continuous_processing_cron.py --mode system_cycle >> \
    /app/logs/continuous_processing_cron.log 2>&1

# 📈 日次レポート（毎日6:00 JST）
0 6 * * * cd /app && export $(cat .env | grep -v '^#' | xargs) && \
    timeout 120 python scripts/cron/daily_report.py >> \
    /app/logs/daily_cron.log 2>&1

# 📊 週次統計・レポート（毎週土曜日 6:00 JST）
0 6 * * 6 cd /app && export $(cat .env | grep -v '^#' | xargs) && \
    timeout 180 python scripts/cron/weekly_report.py >> \
    /app/logs/weekly_cron.log 2>&1

# 📅 週次経済指標配信（毎週日曜日 9:00 JST）
0 9 * * 0 cd /app && export $(cat .env | grep -v '^#' | xargs) && export PYTHONPATH=/app && \
    timeout 300 python scripts/cron/weekly_economic_indicators_discord.py >> \
    /app/logs/weekly_economic_indicators_cron.log 2>&1

# 🗑️ 経済指標キャッシュクリーンアップ（毎日 3:00 JST）
0 3 * * * cd /app && export $(cat .env | grep -v '^#' | xargs) && export PYTHONPATH=/app && \
    timeout 60 python scripts/cron/economic_calendar_cache_manager.py --cleanup >> \
    /app/logs/economic_calendar_cache_cleanup.log 2>&1

# 🔍 システムヘルスチェック（30分間隔）
*/30 * * * * cd /app && export $(cat .env | grep -v '^#' | xargs) && \
    timeout 10 python scripts/monitoring/realtime_monitor.py --interval 1 --no-alerts >> \
    /app/logs/health_cron.log 2>&1

# 🗑️ ログローテーション（毎日2:00 JST）
0 2 * * * cd /app/logs && find . -name "*.log" -size +10M -exec gzip {} \; && \
    find . -name "*.gz" -mtime +7 -delete >> \
    /app/logs/cleanup.log 2>&1

# 🔍 エラーログ監視（10分間隔）
*/10 * * * * [ -f /app/logs/continuous_processing_cron.log ] && \
    tail -10 /app/logs/continuous_processing_cron.log | grep -i "error\|failed" && \
    echo "$(date): Continuous processing errors detected" >> /app/logs/error_alert.log

# 🧪 環境変数テスト（1時間間隔、デバッグ用）
0 * * * * cd /app && export $(cat .env | grep -v '^#' | xargs) && \
    timeout 30 python tests/integration/test_env_loading.py >> \
    /app/logs/env_test_cron.log 2>&1

# 🤖 統合AI分析（2時間間隔、平日8:00-2:00）
0 8,10,12,14,16,18,20,22,0,2 * * 1-5 cd /app && export $(cat .env | grep -v '^#' | xargs) && \
    timeout 180 python scripts/cron/integrated_ai_discord.py >> \
    /app/logs/integrated_ai_cron.log 2>&1

# 🔍 パフォーマンス監視システムテスト（1時間間隔）
0 * * * * cd /app && export $(cat .env | grep -v '^#' | xargs) && \
    timeout 120 python scripts/cron/performance_monitoring_test_cron.py >> \
    /app/logs/performance_monitoring_test_cron.log 2>&1
